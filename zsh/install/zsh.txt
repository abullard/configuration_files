# --- Setup Z

if echo $(uname -s) | grep -q "Linux"; then
    source $HOME/.z.sh
elif echo $(uname -s) | grep -q "Darwin"; then
    source /opt/homebrew/etc/profile.d/z.sh
else
    echo "z is not properly sourced!!"
fi

# --- unalias existing commands
unalias z 2> /dev/null
